import java.util.Scanner;

public class DataTypeDemo {
    public static void main(String[] args) {
        // Create a Scanner object for user input
        Scanner scanner = new Scanner(System.in);

        // Input and output for an integer (int)
        System.out.print("Enter an integer (int): ");
        int intValue = scanner.nextInt();
        System.out.println("You entered the integer: " + intValue);

        // Input and output for a double
        System.out.print("Enter a decimal number (double): ");
        double doubleValue = scanner.nextDouble();
        System.out.println("You entered the double: " + doubleValue);

        // Clear the scanner buffer
        scanner.nextLine();

        // Input and output for a single character (char)
        System.out.print("Enter a single character (char): ");
        char charValue = scanner.nextLine().charAt(0);
        System.out.println("You entered the character: " + charValue);

        // Input and output for a string
        System.out.print("Enter a string: ");
        String stringValue = scanner.nextLine();
        System.out.println("You entered the string: " + stringValue);

        // Input and output for a boolean
        System.out.print("Enter a boolean value (true/false): ");
        boolean booleanValue = scanner.nextBoolean();
        System.out.println("You entered the boolean: " + booleanValue);

        // Close the scanner
        scanner.close();
    }
}
